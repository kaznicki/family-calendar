  let cachedRosterSnapshot: Record<string, string> = {}
  const subscribeRoster = (cb: () => void) => {
    rosterMap.observeDeep(cb)
    return () => rosterMap.unobserveDeep(cb)
  }
  const getRosterSnapshot = () => {
    const fresh = JSON.stringify(rosterMap.toJSON())
    const cached = JSON.stringify(cachedRosterSnapshot)

    return cachedRosterSnapshot
